\subsection{PeriodiДЌne toДЌke} \label{ssec:periodic}

ZaДЌetna toДЌka je \emph{periodiДЌna}, ДЌe obstaja tak \(n \in \NN\), da je \(f^n (x_0) = x_0\). ДЊe je \(n\) najmanjЕЎe tako ЕЎtevilo, pravimo, da ima periodiДЌna toДЌka periodo \(n\), toДЌka pa je \emph{\(n\)-periodiДЌna}. PeriodiДЌna toДЌka je \emph{fiksna}, ДЌe je njena perioda enaka \num{1}. V nadaljevanju bomo preuДЌili lastnosti fiksnih toДЌk.

\begin{definicija}
    Naj bo \(X\) metriДЌni prostor, \(f \colon X \to X\) zvezna in \(x^*\) fiksna toДЌka \(f\). ДЊe obstaja tak \(r > 0\), da na odprti krogli \(B (x^*, r)\) orbita vsake zaДЌetne toДЌke enakomerno konvergira proti \(x^*\), je \(x^*\) \emph{privlaДЌna}. ToДЌka \(x^*\) je \emph{odbojna}, ДЌe obstaja \(r > 0\), da za vsak \(x_0 \in B (x^*, r)\) in \(x_0 \neq x^*\), obstaja \(n \in \NN\), za katerega je \(f^n (x_0) \notin B (x^*, r)\).
\end{definicija}

\begin{definicija}
    Naj bosta \(X\) in \(Y\) metriДЌna prostora ter \(f \colon X \to X\) in \(g \colon Y \to Y\) zvezni. Pravimo, da sta \(f\) in \(g\) \emph{konjugirani}, ДЌe obstaja homeomorfizem \(h \colon X \to Y\), da velja \(h \circ f = g \circ h\). Tedaj pravimo, da \(h\) konjugira \(f\) in \(g\).
\end{definicija}

Konjugiranost je ekvivalenДЌna relacija z lastnostjo, da za vsak \(n \in \NN\) velja \(f^n = h^{-1} \circ g^n \circ h\). Za nas je zanimiva, saj imajo njeni ekvivalenДЌni razredi enako dinamiko, kar nam nakaЕѕe naslednja trditev.

\begin{trditev}
    Naj bosta \(f \colon X \to X\) in \(g \colon Y \to Y\) konjugirani.
    \begin{enumerate}
        \item ДЊe je \(x^* \in X\) fiksna toДЌka \(f\), tedaj je \(y^* = h (x^*) \in Y\) fiksna toДЌka \(g\).
        \item ДЊe je \(x^*\) privlaДЌna fiksna toДЌka \(f\), tedaj je \(y^* = h (x^*)\) privlaДЌna fiksna toДЌka \(g\).
        \item ДЊe je \(x^*\) odbojna fiksna toДЌka \(f\), tedaj je \(y^* = h (x^*)\) odbojna fiksna toДЌka \(g\).
    \end{enumerate}
\end{trditev}

\begin{dokaz}
    Naj bo \(x^*\) fiksna toДЌka \(f\). Tedaj \(g (y^*) = g (g (x^*)) = h (f (x^*)) = h (x^*) = y^*\). ДЊe je \(x^*\) privlaДЌna, tedaj obstaja \(r > 0\), da zaporedje \(f^n (x)\) enakomerno konvergira k \(x^*\) na \(B (x^*, r)\). Ker je \(h^{-1}\) zvezna, obstaja \(s > 0\), tako da za \(B (y^*, s)\) velja \(h^{-1} (y) \in B (x^*, r)\). To pomeni, da zaporedje \(f^n (h^{-1} (y))\) enakomerno konvergira k \(x^*\) na \(B (y^*, s)\). Ker je \(h\) zvezna pa velja
    \[\lim_{n \to \infty} g^n (y) = \lim_{n \to \infty} h \circ f^n \circ h^{-1} (y) = h \prt{\lim_{n \to \infty} f^n \prt{h^{-1} (y)}} = h (x^*) = y^*\]
    enakomerno na \(B (y^*, s)\).

    ДЊe je \(x^*\) odbojna, obstaja \(r > 0\), da za vsak \(x_0 \in B (x^*, r)\) in \(x_0 \neq x^*\), obstaja \(n \in \NN\), za katerega je \(f^n (x_0) \notin B (x^*, r)\). Ker je \(h\) zvezna, je \(h (B (x^*, r))\) odprta okolica \(y^*\). Torej obstaja \(s > 0\), da velja
    \begin{equation} \label{eqn:podmnozica}
        B (y^*, s) \subset h (B (x^*, r)).
    \end{equation}
    Za vsak \(y \in B (y^*, s)\) z \(y \neq y^*\) obstaja \(x \in B (x^*, r)\) z \(x \neq x^*\), da je \(y = h (x)\). Ker je \(x^*\) odbojna, obstaja tak \(n \in \NN\), da \(f^n (x) = f^n (h^{-1} (y)) \notin B (x^*, r)\). A ker je \(f^n (h^{-1} (y)) = h^{-1} (g^n (y))\), sklepamo, da \(h^{-1} (g^n (y)) \notin B (x^*, r)\) in zato \(g^n (y) \notin h (B (x^*, r))\). Po enaДЌbi \eqref{eqn:podmnozica} torej sledi, da \(g^n (y) \notin B (y^*, s)\).
\end{dokaz}

\subsection{Kaos}

Na intuitivni ravni si besedo kaos navadno predstavljamo kot nered ali zmedo. V naravoslovju ima ta pojem veДЌ ne-ekvivalentnih definicij. Mi se bomo naslonili na definicijo ameriЕЎkega matematika Roberta L.~Devaneya iz leta \num{1989} \cite{Devaney_1986}, v kateri je kaos opredeljen prek periodiДЌnih toДЌk in topoloЕЎke tranzitivnosti.

\begin{definicija}[TopoloЕЎka tranzitivnost]
    Naj bo \(X\) metriДЌni prostor in \(f \colon X \to X\) zvezna preslikava. Pravimo, da je \(f\) \emph{topoloЕЎko tranzitivna}, ДЌe za vsaki odprti mnoЕѕici \(U\) in \(V\) obstajata \(z \in U\) in \(n \in \NN \cup \set{0}\), da je \(f^n (z) \in V\).
\end{definicija}

Pravimo, da ima preslikava \(f\) \emph{gosto orbito}, ДЌe obstaja element \(x \in X\), katerega orbita je gosta podmnoЕѕica \(X\).

\begin{trditev} \label{prop:tridevet}
    Naj bo \(X\) metriДЌni prostor in \(f \colon X \to X\) zvezna preslikava. ДЊe ima \(f\) gosto orbito, je topoloЕЎko tranzitivna.
    % ДЊe je \(M\) separabilen in \(\num{2}\)-ЕЎteven, potem topoloЕЎka tranzitivnost implicira gosto orbito.
\end{trditev}

\begin{dokaz}
    Naj bosta \(U, V \subseteq X\) neprazni odprti mnoЕѕici in naj bo \(\zap{x}\) gosta orbita preslikave \(f\). Potem obstajata \(k, m \in \NN\), da velja \(x_k \in U\) in \(x_m \in V\). Brez ЕЎkode za sploЕЎnost predpostavimo \(m \geq k\). Tedaj je \(x_k \in U\) ter \(f^{\, m - k} (x_k) \in V\).
    % Naj bo \(X\) brez izoliranih toДЌk in \(\zap{z}\) gosta orbita. Potem za neprazni odprti mnoЕѕici \(U, V \subseteq M\) obstajata \(x_k \in U\) in \(x_k \in V \setminus \{x_0, x_1, \dots, x_k\}\). Slednja mnoЕѕica je prav tako odprta in neprazna. Ker je \(m > k\) in \(f^{m - k} (U) \cap V \neq \emptyset\), je \(f\) topoloЕЎko tranzitivna.

    % Naj bo \(M\) separabilen in \(\num{2}\)-ЕЎteven ter naj \(f\) nima goste orbite. Naj bo \(\zap{V}\) ЕЎtevna baza prostora. Za vsak \(x \in M\) obstaja \(V_{n (x)}\), tako da za vsak \(k \geq 0\) velja \(f^k (x) \notin V_{n (x)}\). Ker pa je \(f\) topoloЕЎko tranzitivna je
    % \[\bigcup_{k = 0}^{\infty} f^{- k} \prt{V_{n (x)}}\]
    % odprta mnoЕѕica
\end{dokaz}

\begin{definicija}[Devaneyjev kaos]
    Naj bo \(X\) neskonДЌen metriДЌni prostor. Pravimo, da je zvezna preslikava \(f \colon X \to X\) \emph{kaotiДЌna}, ДЌe sta izpolnjena naslednja pogoja.
    \begin{enumerate}
        \item MnoЕѕica periodiДЌnih toДЌk je gosta v mnoЕѕici \(X\).
        \item Preslikava \(f\) je topoloЕЎko tranzitivna.
    \end{enumerate}
\end{definicija}

\begin{zgled}
    Na kroЕѕnici \(S^1 \subset \CC\) definiramo rotacijo za kot \(\theta\):
    \[R_\theta \colon S^1 \to S^1; \qquad R_\theta (z) = e^{i \theta \pi} z.\]
    Naj bo \(\theta_1 = 1\). Potem je vsaka toДЌka periodiДЌna s periodo \(\num{2}\), preslikava pa ni topoloЕЎko  tranzitivna. Naj bo \(\theta_2 = p\), kjer je \(p \in \RR \setminus \QQ\). Potem je preslikava topoloЕЎko tranzitivna, a niti ena toДЌka ni periodiДЌna.
\end{zgled}

\begin{zgled}[Podvojitvena preslikava] \label{ex:double}
    Naj bo \(f \colon [0, 1) \to [0, 1)\) preslikava, podana z
    \[
        f (x) =
        \begin{cases}
            2x     & \text{za } 0 \leq x < \frac{1}{2}  \\
            2x - 1 & \text{za } \frac{1}{2} \leq x < 1.
        \end{cases}
    \]
    Trdimo, da je \(f\) kaotiДЌna. OznaДЌimo \(f_1 (x) = 2 x\) in \(f_2 (x) = 2x - 1\).

    Naj bo \(q \in \NN\) liho ЕЎtevilo in \(A = \set{\frac{1}{q}, \frac{2}{q}, \dots, \frac{q - 1}{q}} \subset [0, 1)\). Ni teЕѕko videti, da je \(f \colon A \to A\) injektivna in ker je \(A\) konДЌna, je tudi bijektivna. Torej je vsak ulomek z lihim imenovalcem periodiДЌna toДЌka. Ker so takЕЎni ulomki gosti v \([0, 1)\), ima \(f\) gosto podmnoЕѕico periodiДЌnih toДЌk.

    Za dokaz topoloЕЎke tranzitivnosti pokaЕѕemo, da ima \(f\) gosto orbito. Pomagamo si z zapisom ЕЎtevila v dvojiЕЎki bazi. Opazimo, da \(f\) ``odreЕѕe'' prvo decimalko ЕЎtevila. Naj bo \(\alpha \in [0, 1)\) tako, da vsebuje vsa konДЌna zaporedja:
    \[\alpha \coloneq 0, \underbrace{0 1}_{\text{ena}} \underbrace{00011011}_{\text{dva}} \underbrace{000 001 010 100 011 101 110 111}_{\text{tri}} \dots_{(2)}\]
    Naj bo \(x = 0{,}d_1 d_2 d_3 \dots_{(2)}\) in \(\varepsilon > 0\). Izberemo tak \(n \in \NN\), da je \(2^{- n} < \varepsilon\). Potem obstaja tak \(k \in \NN\), da se ЕЎtevilo \(f^k (\alpha)\) zaДЌne kot \(0{,}d_1 d_2 \dots d_n d_{n + 1}\), torej da se \(f^k (\alpha)\) in \(x\) zaДЌneta razlikovati kveДЌjemu po \((n + 2)\)-ti decimalki. Potem velja
    \[\abs{f^k (\alpha) - x} \leq \sum_{i = n + 2}^{\infty} \frac{2}{2^{i}} < 2^{- n} < \varepsilon.\]
\end{zgled}

\noindent Za kompleksno eksponentno preslikavo kaotiДЌnost sledi kot posledica izreka \ref{thm:orbits}. Res, goste periodiДЌne toДЌke sledijo po tretji toДЌki izreka, topoloЕЎka tranzitivnost pa zaradi goste orbite po drugi toДЌki. Izrek je prvi dokazal poljski matematik Micha\l\ Misiurewicz leta \num{1981} \cite{Misiurewicz_1981}. S tem je potrdil domnevo, ki jo je leta \num{1926} postavil francoski matematik Pierre J.~L.~Fatou \cite{Fatou_1926}.

Devaney je v svoji prvotni definiciji kaotiДЌnosti zahteval tudi naslednjo lastnost.

\begin{definicija}[ObДЌutljivost na zaДЌetne pogoje]
    Naj bo \((X, d)\) metriДЌni prostor in \(f \colon X \to X\) zvezna preslikava. Pravimo, da je \(f\) obДЌutljiva na zaДЌetne pogoje, ДЌe obstaja \emph{obДЌutljivostna konstanta} \(\delta > 0\), da za vsak \(x \in M\) in vsak \(\varepsilon > 0\) obstaja \(y \in M\), da je \(d (x, y) < \varepsilon\) in \(d (f^N (x), f^N (y)) \geq \delta\) za nek \(N \in \NN\).
\end{definicija}

\noindent To pomeni, da majhna motnja v zaДЌetnih pogojih lahko privede do znatnih sprememb pri iteraciji dinamiДЌnega sistema. Vendar pa se je izkazalo, da te lastnosti pri definiciji kaotiДЌne preslikave v veДЌini primerov ni potrebno zahtevati. Leta \num{1992} so namreДЌ dokazali naslednji izrek \cite{Banks_1992}.

\begin{izrek}
    Naj bo \((X, d)\) neskonДЌen metriДЌni prostor in \(f \colon X \to X\) kaotiДЌna preslikava. Potem je \(f\) obДЌutljiva na zaДЌetne pogoje, razen v primeru, ko je \(X\) sestavljen iz ene periodiДЌne orbite.
\end{izrek}

\begin{dokaz}
    Predpostavimo, da \(X\) ni zgolj periodiДЌna orbita. Ker so periodiДЌne toДЌke goste, obstajata vsaj dve razliДЌni periodiДЌni orbiti. ДЊe bi imeli skupno toДЌko, bi bili enaki. Torej sta disjunktni in zato obstajata periodiДЌni toДЌki \(q, r\), da je
    \[\delta \coloneq \frac{1}{8} \min \set{d (f^n (q), f^m (r)) : n, m \in \NN} > 0.\] Pokazali bomo, da je \(\delta\) obДЌutljivostna konstanta.

    ДЊe je \(x \in X\), je orbita ene od toДЌk \(q, r\) vedno oddaljena od \(x\) za vsaj \(4 \delta\). V nasprotnem primeru, ko bi bile obe v neki toДЌki oddaljene od \(x\) za manj kot \(4 \delta\), bi bila razdalja med orbitama manjЕЎa od \(8 \delta\). Brez ЕЎkode za sploЕЎnost je to orbita \(q\).

    Naj bo \(\varepsilon \in (0, \delta)\). Ker so periodiДЌne toДЌke goste, obstaja periodiДЌna toДЌka \(p \in B (x, \varepsilon)\) s periodo \(n\). Naj bo \(V\) mnoЕѕica toДЌk, ki so po \(n\) iteracijah oddaljene od \(q\) za kveДЌjemu \(\delta\):
    \[V \coloneq \bigcap_{i = 0}^n f^{- i} (B (f^i (q), \delta)).\]
    Po topoloЕЎki tranzitivnosti obstaja \(k \in \NN\), da je \(f^k (B (x, \varepsilon)) \cap V \neq \emptyset\). Torej obstaja \(y \in B (x, \varepsilon)\), da je \(f^k (y) \in V\). ДЊe definiramo \(j \coloneq \lfloor k / n \rfloor + 1\), je \(k / n < j \leq (k / n) + 1\) in
    \[k = n \cdot \frac{k}{n} < n j \leq n \prt{\frac{k}{n} + 1} = k + n.\]
    Torej, ДЌe je \(N \coloneq n j\), je \(0 < N - k \leq n\). Ker je \(f^N (p) = p\), po trikotniЕЎki neenakosti dobimo
    \begin{align}
        d \prt{f^N (p), f^N (y)} & = d \prt{p, f^N (y)} \nonumber                                                              \\
                                 & \geq d \prt{x, f^{N - k} (q)} - d \prt{f^{N - k} (q), f^N (y)} - d (p, x) \label{eqn:ocena} \\
                                 & \geq 4 \delta - \delta - \delta = 2 \delta. \nonumber
    \end{align}
    Tu smo upoЕЎtevali \(p \in B (x, \varepsilon) \subset B (x, \delta)\) in
    \[f^N (y) = f^{N - k} \prt{f^k (y)} \in f^{N - k} (V) \subset B \prt{f^{N - k} (q), \delta},\]
    kar velja po definiciji \(V\). Vemo, da \(p, y \in B (x, \varepsilon)\) in zato po (\ref{eqn:ocena}) velja
    \[d (f^N (p), f^N (x)) \geq \delta \qquad \text{ali} \qquad d (f^N (y), f^N (x)) \geq \delta.\]
\end{dokaz}

\subsection{SimboliДЌna dinamika}

SimboliДЌna dinamika je orodje, ki nam pomaga pri preuДЌevanju dinamiДЌnih sistemov. Z njo smo se sreДЌali Еѕe v zgledu \ref{ex:double}, kjer je bila kljuДЌna za dokaz topoloЕЎke tranzitivnosti.

Naj bo \(N \geq 2\) naravno ЕЎtevilo. Z \(\Sigma_N\) oznaДЌimo mnoЕѕico neskonДЌnih zaporedij \(\overline{s} = (s_0, s_1, s_2, \dots)\), definirano kot
\[\Sigma_N \coloneq \set{\overline{s} = (s_0, s_1, s_2, \dots) : - N \leq s_i \leq N, s_i \in \ZZ}.\]
Na \(\Sigma_N\) definiramo metriko
\[d (\overline{s}, \overline{s}^*) = \sum_{i = 0}^{\infty} \frac{|s_i - s_i^*|}{N^i}.\]

\begin{izrek} \label{thm:blizu}
    Naj bosta \(\overline{s}, \overline{t} \in \Sigma_N\). Potem je \(d(\overline{s}, \overline{t}) < 1/N^n\) natanko tedaj, ko je \(s_i = t_i\) za \(i \leq n\).
\end{izrek}

\begin{dokaz}
    ДЊe je \(s_i = t_i\) za \(i \leq n\), velja
    \[d(\overline{s}, \overline{t}) = \sum_{i = n + 1}^{\infty} \frac{|s_i - t_i|}{N^i} \leq \sum_{i = n + 1}^{\infty} \frac{1}{N^i} = \frac{1}{N^n}.\]
    ДЊe \(s_j \neq t_j\) za nek \(j \leq n\), mora veljati
    \[d(\overline{s}, \overline{t}) \geq \frac{1}{2^j} \geq \frac{1}{N^n}.\]
\end{dokaz}

\noindent Z naslednjo lemo pokaЕѕemo, da za vsak \(\overline{s} \in \Sigma_n\) mnoЕѕice
\[V(\overline{s}, k) = \set{\overline{t} = (t_i) \in \Sigma_N : t_i = s_i, i = 0, 1, \dots, k}\]
za \(k \geq 0\) tvorijo bazo okolic \(\overline{s}\) za topologijo porojeno z zgornjo metriko.

\begin{lema} \label{lem:seq-metric}
    Naj bosta \(\overline{s}, \overline{t} \in \Sigma_N\). Potem veljata naslednji trditvi.
    \begin{enumerate}[label=(\arabic*)]
        \item ДЊe je \(s_i = t_i\) za \(i \leq k\), velja \[d (\overline{s}, \overline{t}) \leq \frac{2}{N^{k - 1} (N - 1)}.\]
        \item ДЊe je \(d (\overline{s}, \overline{t}) < 1 / N^k\), je \(s_i = t_i\) za \(i \leq k\).
    \end{enumerate}
\end{lema}

\begin{dokaz}
    (1) Naj bosta \(\overline{s}, \overline{t} \in \Sigma_N\) taka, da je \(s_i = t_i\) za \(i \leq k\). Potem je
    \[d (\overline{s}, \overline{t}) = \sum_{i \geq k + 1} \frac{|s_i - t_i|}{N^i} \leq 2 N \sum_{i \geq k + 1} \frac{1}{N^i} = \frac{2}{N^{k - 1} (N - 1)}.\]
    (2) ДЊe je \(s_i \neq t_i\) za nek \(i \leq k\), je
    \[d (\overline{s}, \overline{t}) \geq \frac{|s_i - t_i|}{N^i} > \frac{1}{N^i} \geq \frac{1}{N^k}.\]
\end{dokaz}

\noindent Topologija, porojena z zgornjo metriko, je torej ekvivalentna produktni topologiji.

V simboliДЌni dinamiki je pomembna \emph{preslikava zamika}, to je preslikava \(\sigma \colon \Sigma_N \to \Sigma_N\) podana s predpisom:
\[\sigma (s_0, s_1, s_2, \dots) = (s_1, s_2, s_3, \dots).\]
Njena \(n\)-ta iteracija je podana s predpisom
\[\sigma^n (s_0, s_1, s_2, \dots) = (s_n, s_{n + 1}, s_{n + 2}, \dots).\]
IzkaЕѕe se, da je \(\sigma\) zvezna. Res, naj bo \(\varepsilon > 0\). Izberemo tak \(i \in \NN\), da je \(1 / N^i < \varepsilon\), in \(\delta = 1 / N^{n + 1}\). ДЊe sta \(\overline{s}, \overline{t} \in \Sigma_N\) taka, da je \(d(\overline{s}, \overline{t}) < \delta\), mora po izreku \ref{thm:blizu} veljati \(s_i = t_i\) za vsak \(i \leq n + 1\). To pomeni, da se \(\overline{s}\) in \(\overline{t}\) ujemata na prvih \((n + 1)\) mestih. Torej se \(\sigma (\overline{s})\) in \(\sigma (\overline{t})\) ujemata na prvih \(n\) mestih in spet po istem izreku velja \(d (\sigma (\overline{s}), \sigma (\overline{t})) < \varepsilon\).

\begin{definicija}
    Kompaktno popolnoma nepovezano mnoЕѕico brez izoliranih toДЌk imenujemo \emph{Cantorjeva mnoЕѕica}.
\end{definicija}

\begin{trditev}
    MnoЕѕica \(\Sigma_N\) je Cantorjeva.
\end{trditev}

\begin{dokaz}
    DokaЕѕimo, da ni izoliranih toДЌk. Naj bo \(\overline{s} \in \Sigma_N\), \(\varepsilon > 0\) in \(k \in \NN\) tak, da je \(2/N^{k - 1} (N -1) < \varepsilon\). Po lemi \ref{lem:seq-metric}, ДЌe je \(\overline{t} \in \Sigma_N\) tak, da je \(t_i = s_i\) za \(i \leq k\), potem
    \[d (\overline{s}, \overline{t}) \leq \frac{2}{N^{k - 1} (N - 1)} < \varepsilon.\]
    To pomeni, da krogla \(B (\overline{s}, \varepsilon)\) vsebuje neskonДЌno elementov.

    DokaЕѕimo, da je \(\Sigma_N\) popolnoma nepovezana. Recimo, da obstaja povezana \(X \subset \Sigma_N\) in vzamemo \(\overline{s}, \overline{s}^* \in \Sigma_N\) za katera \(\overline{s} \neq \overline{s}^*\). Potem obstaja \(i \in \NN\), tako da \(s_i \neq s_i^*\). Definiramo mnoЕѕici
    \[X_1 \coloneq \set{\overline{t} \in \Sigma_N : t_i = s_i} \qquad \text{in} \qquad X_2 = \set{\overline{t} \in \Sigma_N : t_i \neq s_i}\]
    in opazimo, da je \((X_1 \cap X) \cup (X_2 \cap X_1) = X\), \(X_1 \cap X_2 = \emptyset\), \(\overline{s} \in X_1\) in \(\overline{s}^* \in X_2\). Do protislovja bomo torej priЕЎli, ДЌe pokaЕѕemo, da sta \(X_1\) in \(X_2\) odprti. Ker je
    \[X_2 = \bigcup_{s \neq s_i} \set{\overline{t} \in \Sigma_N : t_i = s},\]
    je dovolj, da pokaЕѕemo, da je \(X_1\) odprta. Za \(\overline{t} \in X_1\) izberemo tak \(\varepsilon\), da bo \(\varepsilon < 1 / N^i\). Po lemi \ref{lem:seq-metric} \(\overline{r} \in B (\overline{t}, \varepsilon)\) implicira \(r_j = t_j\) za \(j \leq i\). Torej \(r_i = s_i\) in zato \(B (\overline{t}, \varepsilon) \subset X_1\).

    Dokaz sklenemo s preprostim razmislekom: iz topologije vemo, da je produkt poljubne druЕѕine kompaktnih prostorov kompakten v produktni topologiji.
\end{dokaz}
